делается RepositoryScope и BlocScope 

class RepositoryScope extends StatelessWidget {
  final Widget child;
  const RepositoryScope({super.key, required this.child,});

  @override
  Widget build(BuildContext context) {
    return MultiRepositoryProvider(
      providers: [
        // UserRepository абстрактный класс который из domain UserRepositoryImpl его реализация тут мы можем менять          реализации чего угодно не меня пол проекта всякие mock репозитории для тестрирования //
        RepositoryProvider<UserRepository>(create: (context) => UserRepositoryImpl()),
      ],
      child: child,
    );
  }
}
